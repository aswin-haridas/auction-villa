# table structure in supabase 


```
create table public."Auction" (
  id uuid not null default gen_random_uuid (),
  name text null,
  image text[] null,
  price numeric null,
  end_time time without time zone null,
  buyout_price numeric null,
  start_time time without time zone null,
  status text null,
  highest_bid numeric null,
  highest_bidder uuid null,
  category text null,
  constraint auctions_pkey primary key (id)
) TABLESPACE pg_default;
```

```
create table public."Bank" (
  id bigint generated by default as identity not null,
  user_id uuid null,
  type text null,
  amount numeric null,
  timestamp timestamp without time zone null,
  constraint Bank_pkey primary key (id),
  constraint Bank_user_id_fkey foreign KEY (user_id) references "User" (user_id)
) TABLESPACE pg_default;
```

```
create table public."Bid" (
  bid_id uuid not null default gen_random_uuid (),
  auction_id uuid null,
  user_id uuid null,
  amount numeric null,
  timestamp timestamp without time zone null,
  constraint Bid_pkey primary key (bid_id),
  constraint Bid_auction_id_fkey foreign KEY (auction_id) references "Auction" (id),
  constraint Bid_user_id_fkey foreign KEY (user_id) references "User" (user_id)
) TABLESPACE pg_default;
```

```
create table public."Painting" (
  painting_id uuid not null default gen_random_uuid (),
  name text null,
  image text[] null,
  owner text null default 'elite'::text,
  acquire_date timestamp without time zone null,
  category text null,
  constraint Paintings_pkey primary key (painting_id)
) TABLESPACE pg_default;
```

```
create table public."Trade" (
  id uuid not null default gen_random_uuid (),
  "auctionId" uuid null default gen_random_uuid (),
  "userId" uuid null default gen_random_uuid (),
  amount numeric null,
  "createdAt" timestamp without time zone null,
  constraint Trades_pkey primary key (id),
  constraint Trades_auctionId_fkey foreign KEY ("auctionId") references "Auction" (id),
  constraint Trades_userId_fkey foreign KEY ("userId") references "User" (user_id)
) TABLESPACE pg_default;
```

```
create table public."User" (
  created_at timestamp without time zone not null,
  username text null,
  password text null,
  user_id uuid not null default gen_random_uuid (),
  balance numeric null default '200000'::numeric,
  constraint auth_pkey primary key (user_id)
) TABLESPACE pg_default;
```
